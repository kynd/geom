import soundMeta from '../../data/sound-meta.json';

export const base = import.meta.env.BASE_URL.replace(/\/$/, '');

export function formatHz(hz) {
  return hz >= 1000 ? `${(hz / 1000).toFixed(1)} kHz` : `${Math.round(hz)} Hz`;
}

<h2>Amplitude and spectrum</h2>
<a href={`${base}/demos/sound-analysis/index.html`} target="_blank" rel="noopener">open full screen ↗</a>
</div>
<div class="demo-embed">
<iframe src={`${base}/demos/sound-analysis/index.html?embedded`} title="Sound Analysis Demo" scrolling="no" allow="autoplay"></iframe>
</div>
<div class="prose">
<p>
  Select a stem from the dropdown and press play.
  The top graph traces the overall RMS amplitude over the last 15 seconds — a running picture of how loud the signal is.
  The bottom chart shows the frequency content of the current frame: 128 Mel-spaced bands, updated at 60 fps.
  The frequency label in the corner shows the active range the analysis detected for that stem.
</p>
<p>
  A Python script analyses each sound file offline before the browser touches it.
  It computes a full-file STFT at 2048-point resolution, then builds a global energy profile across all frequencies.
  Spectral roll-off at 5 % and 85 % of cumulative energy marks where the track actually lives — ignoring sub-bass rumble
  below the content and empty headroom above it.
  A Mel-scale filterbank then maps 128 bands logarithmically across that active range, concentrating resolution
  where the musical content is densest.
  The script processes left and right channels separately and writes one row of numbers per frame at 60 fps:
  L and R RMS amplitude followed by 128 Mel band magnitudes for each channel.
</p>
<p>
  The browser loads that text file and advances through it in lockstep with the audio,
  without doing any signal processing at runtime.
  This matters for video export: every frame's visual state comes from a specific row in the data file,
  so rendering the same frame twice always produces the same result —
  no timing drift, no decoding variance between the audio and the visuals.
</p>
<p>
  The demo skips the silent part at the start of each stem.
  It scans the data file for the first frame where amplitude exceeds a small threshold,
  then cues both the audio and the display to that point.
</p>
</div>
</section>

<div class="prose">
<p>
  Each strip shows a one-minute excerpt — the densest section of each stem, selected by finding the sixty-second window with the highest cumulative amplitude.
  The horizontal axis is time — ten frames collapsed into one pixel column.
  The vertical axis runs from the stem's lowest active frequency at the bottom to its highest at the top.
  Pixel brightness is the peak Mel-band amplitude within that 10-frame window, with a square-root curve applied so quieter detail stays visible.
  L and R channels are shown separately.
</p>
</div>
